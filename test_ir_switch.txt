// Test switch statement with break
int main() {
    int x = 2;
    
    switch (x) {
        case 1:
            print 10;
            break;
        case 2:
            print 20;
            break;
        case 3:
            print 30;
            break;
        default:
            print 0;
            break;
    }
    
    return 0;
}
